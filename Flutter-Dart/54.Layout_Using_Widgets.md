// Layout cheat-sheet

import 'package:flutter/material.dart';

// void main() => runApp(MyApp());
void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Scaffold(
        backgroundColor: Colors.blueGrey[500],
        appBar: AppBar(
          title: Center(
            child: Text('Home'),
          ),
          backgroundColor: Colors.blueGrey[900],
        ),
        body: SafeArea(
          child: Column(
            mainAxisAlignment: MainAxisAlignment.spaceEvenly,
            children: <Widget>[
              Container(
                height: 200,
                width: double.infinity,
                color: Colors.blueGrey[300],
                child: Image(
                  image: NetworkImage(
                      'https://cdn.pilinux.me/images/GoREST/logo/GoREST-Logo.png'),
                ),
              ),
              Container(
                height: 200,
                width: double.infinity,
                color: Colors.blueGrey[300],
                child: Row(
                  mainAxisAlignment: MainAxisAlignment.center,
                  children: <Widget>[
                    Container(
                      height: 180,
                      width: 180,
                      child: Image(
                        image: NetworkImage(
                            'https://cdn.pilinux.me/images/GoREST/logo/GoREST-Logo.png'),
                      ),
                    ),
                    Container(
                      height: 120,
                      width: 120,
                      child: Image(
                        image: NetworkImage(
                            'https://cdn.pilinux.me/images/GoREST/logo/GoREST-Logo.png'),
                      ),
                    ),
                  ],
                ),
              ),
              Container(
                height: 200,
                width: double.infinity,
                color: Colors.blueGrey[300],
                child: Column(
                  mainAxisAlignment: MainAxisAlignment.center,
                  children: <Widget>[
                    CircleAvatar(
                      radius: 70.0,
                      backgroundColor: Colors.blueGrey[200],
                      backgroundImage: NetworkImage(
                          'https://cdn.pilinux.me/images/GoREST/logo/GoREST-Logo.png'),
                    ),
                    Container(
                      margin: EdgeInsets.symmetric(
                        vertical: 6,
                      ),
                      child: Row(
                        mainAxisAlignment: MainAxisAlignment.center,
                        children: <Widget>[
                          Icon(
                            Icons.bookmarks,
                            color: Colors.blueGrey.shade700,
                          ),
                          SizedBox(
                            width: 6,
                          ),
                          Text(
                            'Demo App',
                            style: TextStyle(
                              fontSize: 16,
                            ),
                          ),
                        ],
                      ),
                    ),
                  ],
                ),
              ),
            ],
          ),
        ),
      ),
    );
  }
}
